KatanaListView

This is a realization of the ListView and GridView LAC library contained more features, support for custom add animation。

Maven

<dependency>
    <groupId>net.soulwolf.widget</groupId>
    <url>https://dl.bintray.com/soulwolf/maven</url>
    <artifactId>KatanaListView</artifactId>
    <version>1.0.0</version>
</dependency>

Gradle

allprojects {
   repositories {
      jcenter()
      maven{
        url "https://dl.bintray.com/soulwolf/maven"
      }
   }
}

compile 'net.soulwolf.widget:KatanaListView:1.0.0'
